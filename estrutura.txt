src/
├── assets/              # Imagens, SVGs e ícones globais
├── components/          # Componentes globais e genéricos (Botões, Inputs, Cards)
│   ├── ui/              # Componentes de UI de terceiros ou base (ex: Shadcn ou autorais)
│   └── common/          # Componentes reutilizáveis em ambos os modos (Loading, Modal)
├── features/            # Lógica de negócio dividida por funcionalidades
│   ├── auth/            # Login, Cadastro, Esqueci a senha (Zod schemas aqui)
│   ├── cart/            # Lógica do carrinho de compras
│   ├── checkout/        # Processo de finalização de compra
│   └── products/        # Listagem, Filtros, Detalhes do produto
├── hooks/               # Custom hooks (ex: useAuth, useLocalStorage)
├── layouts/             # Wrappers estruturais
│   ├── UserLayout.jsx   # Header KaBuM!, Footer, etc.
│   └── AdminLayout.jsx  # Sidebar lateral, Dashboard, etc.
├── pages/               # Apenas a declaração das rotas (importam as features)
│   ├── user/            # Home, ProductPage, Profile
│   └── admin/           # Dashboard, Inventory, OrdersManagement
├── services/            # Configurações de API (Axios/Fetch)
├── store/               # Gerenciamento de estado (Context API ou Zustand)
├── styles/              # CSS Global, Variáveis e Resets
├── utils/               # Helpers e Validações (Ex: formatar moeda, schemas do Zod)
└── main.jsx             # Ponto de entrada
